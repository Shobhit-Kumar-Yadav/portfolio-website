4. Find the smallest number from the array

# Define an array
array=(3 5 1 9 2)

# Find the smallest number
min=${array[0]}
for num in "${array[@]}"; do
  if (( num < min )); then
    min=$num
  fi
done

echo "The smallest number is: $min"

7. Check whether the number is a palindrome or not

# Function to check if a number is a palindrome
is_palindrome() {
  local num=$1
  local reversed_num=$(echo "$num" | rev)
  
  if [ "$num" -eq "$reversed_num" ]; then
    echo "$num is a palindrome"
  else
    echo "$num is not a palindrome"
  fi
}

# Replace 121 with your number
number=121
is_palindrome "$number"
